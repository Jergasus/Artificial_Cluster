65 void Procesador::recalcular_huecos() {
66  huecos = map<int, set<int>>();
67  map<int, Proceso>::const_iterator it = memoria.begin();
68  if (it == memoria.end()) huecos[espacio_total].insert(0);
69  else if (it->first != 0) huecos[it->first].insert(0);
70  while (it != memoria.end()) {
71  int pos_siguiente_proceso = siguiente_proceso(it);

73  if (tamano_hueco > 0) huecos[tamano_hueco].insert(it->first + it->second.consultar_memoria());
